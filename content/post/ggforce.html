--- title: "Zoom Your Plots with ggforce" author: Thiyanga Talagala date: '2022-03-28' categories: [R, tidyverse, ggplot2, scatterplot] tags: [R, tidyverse, ggplot2] summary: "" ---

To enlarge a section of data in your ggplots, use the facet zoom function in the ggforce package. To illustrate I am using the gapminder dataset in gapminder package.

Load necessary

library(tidyverse)
library(ggforce)
library(gapminder)

Zoom according to a continuous scale variable

ggplot(gapminder, aes(y=lifeExp, x=gdpPercap)) + geom_point() + coord_fixed() + 
  facet_zoom(ylim = c(60, 80)) 

ggplot(gapminder, aes(y=lifeExp, x=gdpPercap)) + geom_point() + coord_fixed() + 
  facet_zoom(xlim = c(0, 10000)) 

ggplot(gapminder, aes(y=lifeExp, x=gdpPercap)) + geom_point() + coord_fixed() + 
  facet_zoom(xlim = c(0, 10000), ylim=c(60, 80))

Zoom according to a qualitative variable

ggplot(gapminder, aes(y=lifeExp, x=gdpPercap, col=continent)) + geom_point() + coord_fixed() + 
  facet_zoom(y = continent == 'Oceania') 

ggplot(gapminder, aes(y=lifeExp, x=gdpPercap, col=continent)) + geom_point() + coord_fixed() + 
  facet_zoom(x = continent == 'Oceania') 

ggplot(gapminder, aes(y=lifeExp, x=gdpPercap, col=continent)) + geom_point() + coord_fixed() + 
  facet_zoom(xy = continent == 'Oceania') 

ggplot(gapminder, aes(y=lifeExp, x=gdpPercap, col=continent)) + geom_point() + coord_fixed() + 
  facet_zoom(xy = continent == 'Oceania', split = TRUE)
